(defmacro case [base-value &amp; patterns] 
 (cons 
  'cond 
  (fold (fn [[val &amp; more] res] 
         (concat res (list (if (= val :else) val `(= ~base-value ~val)) (first more)))) ( 
    (partition 2 patterns)))))
